### 快速实现（30分钟）

#### 1. git clone
```python
def clone(self, url: str, path: str, callback=None):
    """克隆远程仓库"""
    self.operationStarted.emit("正在克隆仓库...")
    
    args = ['clone', url, path, '--progress']
    
    def on_finished(success: bool, stdout: str, stderr: str):
        if success:
            msg = f"克隆成功: {path}"
        else:
            msg = stderr or "克隆失败"
        self.operationFinished.emit(success, msg)
        if callback:
            callback(success, msg)
    
    self._run_git_async(args, on_finished, timeout=120)  # 克隆可能很慢
```
