### 1. 强制填写验证

```python
def validateCurrentPage(self, index: int) -> bool:
    """验证当前页面是否可以继续
    
    重写GuideWindow的验证方法实现强制填写
    """
    # 第2页 → 第3页：验证远程仓库信息
    if index == 2:
        if not self.remoteInfoPage.is_valid():
            InfoBar.warning(
                "信息不完整",
                "请填写完整的远程仓库名称和URL才能继续",
                ...
            )
            return False  # 阻止继续
        return True
    
    # 第3页 → 第4页：验证分支配置
    elif index == 3:
        if not self.branchConfigPage.is_valid():
            InfoBar.warning(
                "信息不完整",
                "请选择本地分支并填写远程分支名称才能继续",
                ...
            )
            return False  # 阻止继续
        return True
```

### 2. 实时验证反馈

```python
def _validate_inputs(self):
    """实时验证输入"""
    name = self.nameEdit.text().strip()
    url = self.urlEdit.text().strip()
    
    if not name and not url:
        self.validationLabel.setText("❌ 请填写远程名称和URL")
        self.validationLabel.setTextColor(红色)
        return False
    elif not name:
        self.validationLabel.setText("❌ 请填写远程名称")
        self.validationLabel.setTextColor(红色)
        return False
    elif not url:
        self.validationLabel.setText("❌ 请填写远程URL")
        self.validationLabel.setTextColor(红色)
        return False
    elif name in self.existing_remotes:
        self.validationLabel.setText(f"⚠️ 远程名称 '{name}' 已存在，将会覆盖")
        self.validationLabel.setTextColor(橙色)
        return True
    else:
        self.validationLabel.setText("✅ 信息填写完整")
        self.validationLabel.setTextColor(绿色)
        return True
```

### 3. 异步配置执行

```python
def _on_finish(self):
    """完成配置"""
    # 获取配置信息
    remote_name, remote_url = self.remoteInfoPage.get_remote_info()
    local_branch, remote_branch = self.branchConfigPage.get_branch_config()
    
    def do_config():
        """执行配置操作"""
        # 1. 添加/更新远程仓库
        if remote_name in self.existing_remotes:
            gitService.set_remote_url(remote_name, remote_url)
        else:
            gitService.add_remote(remote_name, remote_url)
        
        # 2. 设置分支跟踪
        gitService.set_upstream(local_branch, remote_name, remote_branch)
        
        # 3. 验证连接（fetch）
        gitService.fetch(remote=remote_name)
        
        return True
    
    # 异步执行，带进度环
    AsyncTask.run(
        func=do_config,
        on_success=on_success,
        on_error=on_error,
        progress_title='正在配置',
        progress_content='正在配置远程仓库和分支跟踪关系...',
        parent=self.parent()
    )
```
